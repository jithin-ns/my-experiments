function useCurrentRoutePermissions() {
  const location = useLocation();

  for (const key in paths) {
    const { url, permissions } = paths[key as keyof typeof paths];

    // matchPath(pattern, pathname)
    if (matchPath({ path: url, end: true }, location.pathname)) {
      return permissions;
    }
  }

  // No match found â†’ no permissions
  return [];
}
